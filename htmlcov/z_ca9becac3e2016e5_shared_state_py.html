42 def _generate_id(self, prefix: str, current_dict_or_list: Any) -> str: 

43 """Génère un ID simple basé sur la taille actuelle.""" 

44 index = 0 

45 try: 

46 if isinstance(current_dict_or_list, (dict, list)): 

47 index = len(current_dict_or_list) 

48 else: 

49 index = 0 

50 state_logger.warning(f"_generate_id: Type inattendu '{type(current_dict_or_list)}' pour prefix '{prefix}'. Index sera 0.") 

51 except Exception as e: 

52 state_logger.error(f"Erreur dans _generate_id pour prefix '{prefix}': {e}", exc_info=True) 

53 index = 999 

54 safe_index = min(index, 9999) 

55 return f"{prefix}_{safe_index + 1}"

57 def add_task(self, description: str) -> str: 

58 """Ajoute une tâche d'analyse et retourne son ID.""" 

59 task_id = self._generate_id("task", self.analysis_tasks)

73 def add_fallacy(self, fallacy_type: str, justification: str, target_arg_id: Optional[str] = None) -> str: 

74 """Ajoute un sophisme identifié et retourne son ID.""" 

75 fallacy_id = self._generate_id("fallacy", self.identified_fallacies) 

76 entry = {"type": fallacy_type, "justification": justification} 

77 log_target_info = "" 

78 if target_arg_id: 

79 if target_arg_id not in self.identified_arguments: 

80 state_logger.warning(f"ID argument cible '{target_arg_id}' pour sophisme '{fallacy_id}' non trouvé dans les arguments identifiés ({list(self.identified_arguments.keys())}).") 

81 entry["target_argument_id"] = target_arg_id 

82 log_target_info = f" (cible: {target_arg_id})" 

83 self.identified_fallacies[fallacy_id] = entry 

84 state_logger.info(f"Sophisme ajouté: {fallacy_id} - Type: {fallacy_type}{log_target_info}") 

85 state_logger.debug(f"État fallacies après ajout {fallacy_id}: {self.identified_fallacies}") 

86 return fallacy_id

88 def add_belief_set(self, logic_type: str, content: str) -> str: 

89 """Ajoute un belief set formel et retourne son ID.""" 

90 normalized_type = logic_type.strip().lower().replace(" ", "_") 

91 bs_id = self._generate_id(f"{normalized_type}_bs", self.belief_sets) 

92 self.belief_sets[bs_id] = {"logic_type": logic_type, "content": content} 

93 state_logger.info(f"Belief Set ajouté: {bs_id} - Type: {logic_type}") 

94 state_logger.debug(f"État belief_sets après ajout {bs_id}: {self.belief_sets}") 

95 return bs_id

97 def log_query(self, belief_set_id: str, query: str, raw_result: str) -> str: 

98 """Enregistre une requête formelle et son résultat brut.""" 

99 log_id = self._generate_id("qlog", self.query_log) 

100 if belief_set_id not in self.belief_sets: 

101 state_logger.warning(f"ID Belief Set '{belief_set_id}' pour query log '{log_id}' non trouvé dans les belief sets ({list(self.belief_sets.keys())}).") 

102 log_entry = {"log_id": log_id, "belief_set_id": belief_set_id, "query": query, "raw_result": raw_result} 

103 self.query_log.append(log_entry) 

104 state_logger.info(f"Requête loggée: {log_id} (sur BS: {belief_set_id}, Query: '{query[:60]}...')") 

105 state_logger.debug(f"État query_log après ajout {log_id} (taille: {len(self.query_log)}): {self.query_log}") 

106 return log_id

202 def to_json(self, indent: Optional[int] = 2) -> str: 

203 """Sérialise l'état actuel en chaîne JSON.""" 

204 state_dict = {k: v for k, v in self.__dict__.items() if not callable(v) and not k.startswith("_logger")} 

205 try: 

206 return json.dumps(state_dict, indent=indent, ensure_ascii=False, default=str) 

207 except TypeError as e: 

208 state_logger.error(f"Erreur de sérialisation JSON de l'état: {e}") 

209 safe_dict = {k: repr(v) for k, v in state_dict.items()} 

210 return json.dumps({"error": f"JSON serialization failed: {e}", "safe_state_repr": safe_dict}, indent=indent)
